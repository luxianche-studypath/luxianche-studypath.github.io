## theory

+ 虚拟机

  **虚拟机**（**VM**）是[虚拟化](https://en.wikipedia.org/wiki/Virtualization)/[仿真](https://en.wikipedia.org/wiki/Emulator)一个的[计算机系统](https://en.wikipedia.org/wiki/Computer_system)。虚拟机基于[计算机体系结构，](https://en.wikipedia.org/wiki/Computer_architectures)并提供物理计算机的功能。它们的实现可能涉及专用的硬件，软件或其组合。

  使用虚拟机可以实现编程一次，在多个机器中运行，比如说[JVM](https://en.wikipedia.org/wiki/Java_virtual_machine)

  还可以保护机器环境，隔离未知程序，比如说[以太坊虚拟机](https://ethdocs.org/en/latest/introduction/what-is-ethereum.html#ethereum-virtual-machine)

+ 在这里使用c语言实现一个虚拟机[LC3](https://en.wikipedia.org/wiki/Little_Computer_3), 并运行汇编语言编写的[2048游戏](https://justinmeiners.github.io/lc3-vm/supplies/2048.obj)

+ `JUST READ THE INSTRUCTION`[LC3文档](https://justinmeiners.github.io/lc3-vm/supplies/lc3-isa.pdf)

  ```c
  #include <stdio.h>
  #include <stdlib.h>
  #include <stdint.h>
  #include <string.h>
  #include <signal.h>
  #include <unistd.h>
  #include <fcntl.h>
  #include <sys/time.h>
  #include <sys/types.h>
  #include <sys/termios.h>
  #include <sys/mman.h>
  
  //Registers
  //"workbench of CPU"
  //addressed by a 16-bit unsigned integer
  //8 general purpose registers(R0-R7)
  //  perform any program calculations
  //1 program counter(PC) register:
  //  the address of the next instruction in memory to execute
  //1 condition falgs(COND) register
  //  information about the previous calculation
  enum
  {
      R_R0 = 0,
      R_R1,
      R_R2,
      R_R3,
      R_R4,
      R_R5,
      R_R6,
      R_R7,
      R_PC,
      R_COND,
      R_COUNT
  };
  
  //Instruction Set
  //command which tell CPU to do fundamental task
  enum
  {
      OP_BR = 0, //branch
      OP_ADD,    //add
      OP_LD,     //load
      OP_ST,     //store
      OP_JSR,    //jump register
      OP_AND,    //bitwise and
      OP_LDR,    //load register
      OP_STR,    //store register
      OP_RTI,    //unused
      OP_NOT,    //bitwise not
      OP_LDI,    //load indirect
      OP_STI,    //store indirect
      OP_JMP,    //jump
      OP_RES,    //reserved (unused)
      OP_LEA,    //load effective address
      OP_TRAP    //execute trap
  };
  
  //Trap Routines
  //system calls
  enum
  {
      TRAP_GETC = 0x20,  //get character from keyboard, not echoed onto the terminal
      TRAP_OUT = 0x21,   //output a character
      TRAP_PUTS = 0x22,  //output a word string
      TRAP_IN = 0x23,    //get character from keyboard, echoed onto the terminal
      TRAP_PUTSP = 0x24, //output a byte string
      TRAP_HALT = 0x25   //halt the program
  };
  
  //Condition Flags
  //R_COND register stores condition flags
  enum
  {
      FL_POS = 1 << 0, //P
      FL_ZRO = 1 << 1, //Z
      FL_NEG = 1 << 2  //N
  };
  
  //Memory Mapped Registers
  //KBSR and KBDR allows you to poll the state of the device and continue execution,
  //so the program can stay responsive while waiting for input.
  enum
  {
      MR_KBSR = 0xFE00, //keyboard status
      MR_KBDR = 0xFE02  //keyboard data
  };
  
  //Memory
  //65,536 memory locations
  //addressed by a 16-bit unsigned integer
  //each location stores a 16-bit value
  //store a total of 128kb
  uint16_t memory[UINT16_MAX];
  
  uint16_t reg[R_COUNT];
  
  uint16_t sign_extend(uint16_t x, int bit_count)
  {
      if ((x >> (bit_count - 1)) & 1) //determine +/-:if x is negative
      {
          //toggle to 1
          x |= (0xFFFF << bit_count);
      };
      return x;
  }
  
  uint16_t swap16(uint16_t x)
  {
      //deal with big-endian
      return (x << 8) | (x >> 8);
  }
  
  void update_flags(uint16_t r)
  {
      //The condition codes are set
      //based on whether the result is negative, zero, or positive
      if (reg[r] = 0)
      {
          reg[R_COND] = FL_ZRO;
      }
      else if (reg[r] >> 15)
      {
          //negative number
          reg[R_COND] = FL_NEG;
      }
      else
      {
          reg[R_COND] = FL_POS;
      }
  }
  
  void read_image_file(FILE *file)
  {
      //the origin tells us where in memory to place the image
      uint16_t origin;
      fread(&origin, sizeof(origin), 1, file);
      origin = swap16(origin);
  
      //we know the maximum file size so we only need one fread
      uint16_t max_read = UINT16_MAX - origin;
      uint16_t *p = memory + origin;
      size_t read = fread(p, sizeof(uint16_t), max_read, file);
  
      //swap to little endian
      while (read-- > 0)
      {
          *p = swap16(*p);
          ++p;
      }
  }
  
  int read_image(const char *image_path)
  {
      FILE *file = fopen(image_path, "rb");
      if (!file)
          return 0;
      read_image_file(file);
      fclose(file);
      return 1;
  }
  
  /* Check Key */
  uint16_t check_key()
  {
      fd_set readfds;
      FD_ZERO(&readfds);
      FD_SET(STDIN_FILENO, &readfds);
  
      struct timeval timeout;
      timeout.tv_sec = 0;
      timeout.tv_usec = 0;
      return select(1, &readfds, NULL, NULL, &timeout) != 0;
  }
  
  /* Memory Access */
  void mem_write(uint16_t address, uint16_t val)
  {
      memory[address] = val;
  }
  
  uint16_t mem_read(uint16_t address)
  {
      if (address == MR_KBSR)
      {
          if (check_key())
          {
              memory[MR_KBSR] = (1 << 15);
              memory[MR_KBDR] = getchar();
          }
          else
          {
              memory[MR_KBSR] = 0;
          }
      }
      return memory[address];
  }
  
  /* Input Buffering */
  struct termios original_tio;
  
  void disable_input_buffering()
  {
      tcgetattr(STDIN_FILENO, &original_tio);
      struct termios new_tio = original_tio;
      new_tio.c_lflag &= ~ICANON & ~ECHO;
      tcsetattr(STDIN_FILENO, TCSANOW, &new_tio);
  }
  
  void restore_input_buffering()
  {
      tcsetattr(STDIN_FILENO, TCSANOW, &original_tio);
  }
  
  /* Handle Interrupt */
  void handle_interrupt(int signal)
  {
      restore_input_buffering();
      printf("\n");
      exit(-2);
  }
  
  //Produce
  //1. Load one instruciont from memory at the address of the PC register
  //2. PC++
  //3. check opcode to determine which type of instruciton it should perform
  //4. Perform the instruction using the paramerters in the instruction
  //5. Go back to step 1
  int main(int argc, const char *argv[])
  {
      //load arguments
      if (argc < 2)
      {
          //usage
          puts("lc3_vm [image-file1] ...");
          exit(2);
      }
      for (int i = 1; i < argc; i++)
      {
          if (!read_image(argv[i]))
          {
              printf("falied to load image: %s\n", argv[i]);
              exit(1);
          }
      }
      //setup
      signal(SIGINT, handle_interrupt);
      disable_input_buffering();
  
      //set pc to starting position
      //0x3000 if the default
      enum
      {
          PC_START = 0x3000
      };
      reg[R_PC] = PC_START;
  
      int running = 1;
      while (running)
      {
          //FETCH
          uint16_t instr = mem_read(reg[R_PC]++);
          uint16_t op = instr >> 12;
  
          switch (op)
          {
          case OP_ADD:
              //add
              {
                  uint16_t r0 = (instr >> 9) & 0x7;
                  uint16_t r1 = (instr >> 6) & 0x7;
                  uint16_t imm_flag = (instr >> 5) & 0x1; //is negative
                  if (imm_flag)
                  {
                      uint16_t imm5 = sign_extend(instr & 0x1F, 5);
                      reg[r0] = reg[r1] + imm5;
                  }
                  else
                  {
                      uint16_t r2 = instr & 0x7;
                      reg[r0] = reg[r1] + reg[r2];
                  }
                  update_flags(r0);
              }
              break;
          case OP_AND:
              //and
              {
                  uint16_t r0 = (instr >> 9) & 0x7;
                  uint16_t r1 = (instr >> 6) & 0x7;
                  uint16_t imm_flag = (instr >> 5) & 0x1;
                  if (imm_flag)
                  {
                      uint16_t imm5 = sign_extend(instr & 0x1F, 5);
                      reg[r0] = imm5 & r1;
                  }
                  else
                  {
                      uint16_t r2 = instr & 0x7;
                      reg[r0] = reg[r1] & reg[r2];
                  }
                  update_flags(r0);
              }
              break;
          case OP_NOT:
              //not
              {
                  uint16_t r0 = (instr >> 9) & 0x7;
                  uint16_t r1 = (instr >> 6) & 0x7;
                  reg[r0] = ~reg[r1];
                  update_flags(r0);
              }
              break;
          case OP_BR:
              //branch
              {
                  uint16_t pc_offset = sign_extend(instr & 0x1FF, 9);
                  uint16_t cond_flag = (instr >> 9) & 0x7;
                  if (cond_flag & reg[R_COND])
                  {
                      reg[R_PC] += pc_offset;
                  }
              }
              break;
          case OP_JMP:
              //jump
              {
                  uint16_t r1 = (instr >> 6) & 0x7;
                  reg[R_PC] = reg[r1];
              }
              break;
          case OP_JSR:
              //jump register
              {
                  reg[R_R7] = reg[R_PC];
                  if ((instr >> 11) & 0x1)
                  {
                      uint16_t pc_offset = sign_extend(instr & 0x7FF, 11);
                      reg[R_PC] += pc_offset;
                  }
                  else
                  {
                      uint16_t r1 = (instr >> 6) & 0x7;
                      reg[R_PC] = reg[r1];
                  }
              }
              break;
          case OP_LD:
              //load
              {
                  uint16_t r0 = (instr >> 9) & 0x7;
                  uint16_t pc_offset = sign_extend(instr & 0x1FF, 9);
                  reg[r0] = mem_read(reg[R_PC] + pc_offset);
                  update_flags(r0);
              }
              break;
          case OP_LDI:
              //load indirect
              //destination register
              {
                  uint16_t r0 = (instr >> 9) & 0x7;
                  //PCoffet
                  uint16_t pc_offset = sign_extend(instr & 0x1FF, 9);
                  reg[r0] = mem_read(mem_read(reg[R_PC] + pc_offset));
                  update_flags(r0);
              }
              break;
          case OP_LDR:
              //load register
              {
                  uint16_t r0 = (instr >> 9) & 0x7;
                  uint16_t r1 = (instr >> 6) & 0x7;
                  uint16_t offset = sign_extend(instr & 0x3F, 6);
                  reg[r0] = mem_read(reg[r1] + offset);
                  update_flags(r0);
              }
              break;
          case OP_LEA:
              //load effective address
              {
                  uint16_t r0 = (instr >> 9) & 0x7;
                  uint16_t pc_offset = sign_extend(instr & 0x1FF, 9);
                  reg[r0] = reg[R_PC] + pc_offset;
                  update_flags(r0);
              }
              break;
          case OP_ST:
              //store
              {
                  uint16_t r0 = (instr >> 9) & 0x7;
                  uint16_t pc_offset = sign_extend(instr & 0x1FF, 9);
                  mem_write(reg[R_PC] + pc_offset, reg[r0]);
              }
              break;
          case OP_STI:
              //store inderect
              {
                  uint16_t r0 = (instr >> 9) & 0x7;
                  uint16_t pc_offset = sign_extend(instr & 0x1FF, 9);
                  mem_write(mem_read(reg[R_PC] + pc_offset), reg[r0]);
              }
              break;
          case OP_STR:
              //store register
              {
                  uint16_t r0 = (instr >> 9) & 0x7;
                  uint16_t r1 = (instr >> 6) & 0x7;
                  uint16_t offset = sign_extend(instr & 0x3F, 6);
                  mem_write(reg[r1] + offset, reg[r0]);
              }
              break;
          case OP_TRAP:
              //execute trap
              {
                  uint16_t trapvect = instr & 0xFF;
                  switch (trapvect)
                  {
                  case TRAP_GETC:
                      reg[R_R0] = (uint16_t)getchar();
                      break;
                  case TRAP_OUT:
                      putchar((char)reg[R_R0]);
                      fflush(stdout);
                      break;
                  case TRAP_PUTS:
                  {
                      uint16_t *ch = memory + reg[R_R0];
                      while (*ch++)
                          putc((char)*ch, stdout);
                      fflush(stdout);
                  }
                  break;
                  case TRAP_IN:
                  {
                      printf("Enter a character: ");
                      char ch = getchar();
                      putc(ch, stdout);
                      reg[R_R0] = (uint16_t)ch;
                  }
                  break;
                  case TRAP_PUTSP:
                  {
                      uint16_t *ch = memory + reg[R_R0];
                      while (*ch++)
                      {
                          char char1 = (*ch) & 0xFF;
                          putc(char1, stdout);
                          char char2 = (*ch) >> 8;
                          putc(char2, stdout);
                      }
                      fflush(stdout);
                  }
                  break;
                  case TRAP_HALT:
                  {
                      puts("HALT");
                      fflush(stdout);
                      running = 0;
                  }
                  break;
                  default:
                      break;
                  }
              }
              break;
          case OP_RES:
          case OP_RTI:
          default:
              //bad opcode
              abort();
              break;
          }
      }
      //shutdown
      restore_input_buffering();
  
  ```
